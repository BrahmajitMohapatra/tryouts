22:     def run(block, context=nil, &inline)
23:       # A Proc object takes precedence over an inline block. 
24:       runtime = (block.nil? ? inline : block)
25:       response = Tryouts::Drill::Reality.new
26:       begin
27:         if runtime.nil?
28:           ret = @rbox.send *@rbox_args
29:         else
30:           ret = @rbox.instance_eval &runtime
31:         end
32:         response.ecode = ret.exit_code
33:         if ret.stdout.size == 1
34:           response.output = ret.stdout.first 
35:         else
36:           response.output = Array.new(ret.stdout)  # Cast the Rye::Rap object
37:         end
38:         response.error = ret.stderr unless ret.stderr.empty?
39:       rescue Rye::CommandNotFound => ex
40:         puts ex.message, ex.backtrace if Tryouts.debug? && Tryouts.verbose > 2
41:         response.etype = ex.class
42:         response.ecode = ex.exit_code
43:         response.error = "[#{@rbox.host}] Command not found: #{ex.message}"
44:         response.trace = ex.backtrace
45:       rescue Rye::CommandError => ex
46:         puts ex.message, ex.backtrace if Tryouts.debug? && Tryouts.verbose > 2
47:         response.etype = ex.class
48:         response.ecode = ex.exit_code
49:         response.output = ex.stdout
50:         response.error = ex.stderr.join $/
51:         response.trace = ex.backtrace
52:       end
53:       response
54:     end
